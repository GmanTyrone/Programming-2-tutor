#include <sstream>
class Convert
{
public:
  static string toString(int n)
  {
    std::stringstream ss;
    ss << n;
    return ss.str();
  }
  static string toString(char c)
  {
    std::stringstream ss;
    ss << c;
    return ss.str();
  }
  static string toString(double d)
  {
    std::stringstream ss;
    ss << d;
    return ss.str();
  }
  static string toString(Point p)
  {
    std::stringstream ss;
    ss << "(" << p.x << ", "<< p.y << ")";
    return ss.str();
  }
};
